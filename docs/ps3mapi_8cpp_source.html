285String PS3Mapi::_sendCommand(const String& apiArg, const String& route) {
286 HTTPClient http;
287 String url = "http://" + ip + "/" + apiArg + "?" + urlEncode(route);
288 http.begin(url);
289
290 int httpResponseCode = http.GET();
291 String value = "";
292 if (httpResponseCode > 0) {
293 value = extractResponse(http.getString());
294 } else Serial.println("Failed to connect to the specified IP. Is the IP correct, and is the system running?");
295 http.end();
296 return value;
297}
307std::vector<String> PS3Mapi::_sendCommandArray(const String& apiArg, const String& route) {
308 HTTPClient http;
309 String url = "http://" + ip + "/" + apiArg + "?" + urlEncode(route);
310 http.begin(url);
311
312 int httpResponseCode = http.GET();
313 std::vector<String> result;
314 if (httpResponseCode < 0) Serial.println("Failed to connect to the specified IP. Is the IP correct, and is the system running?");
315 result = arrayToList(http.getString());
316 http.end();
317 return result;
318}
335bool PS3Mapi::_sendCommandBool(const String& apiArg, const String& route) {
336 HTTPClient http;
337 String url = "http://" + ip + "/" + apiArg;
338 if (route.length() > 0) url += "?" + urlEncode(route);
339 Serial.println("URL " + url);
340
341 http.begin(url);
342 int httpResponseCode = http.GET();
343 if(httpResponseCode < 0) Serial.println("Failed to connect to the specified IP. Is the IP correct, and is the system running?");
344
345 String respStr = extractResponse(http.getString());
346 http.end();
347
348 if (respStr == "0" || respStr == "false") return false;
349 if (respStr == "1" || respStr == "true") return true;
350
351
352 return httpResponseCode > 0; //fallback to simple http response code check
353}

298 String result = "";
299

307bool PS3Mapi::Pad::holdButtons(const std::vector<String>& buttons, float releaseAfter) {
308 String result = "";
309
310 for (size_t i = 0; i < buttons.size(); ++i) {
311 result += buttons[i];
312 if (i < buttons.size() - 1) {
313 result += "_";
314 }
315 }
316 ps3mapi->_sendCommandBool("pad.ps3", "hold" + result);
317
318 if(releaseAfter > 0.0f) delay(releaseAfter);
319
320 return PS3Mapi::Pad::releaseButtons();
321}

409String PS3Mapi::_sendCommand(const String& apiArg, const String& route) {
410 HTTPClient http;
411 String url = "http://" + ip + "/" + apiArg + "?" + urlEncode(route);
412 http.begin(url);
413
414 int httpResponseCode = http.GET();
415 String value = "";
416 if (httpResponseCode > 0) {
417 value = extractResponse(http.getString());
418 } else Serial.println("Failed to connect to the specified IP. Is the IP correct, and is the system running?");
419 http.end();
420 return value;
421}

434 http.begin(url);
435
436 int httpResponseCode = http.GET();

442}
459bool PS3Mapi::_sendCommandBool(const String& apiArg, const String& route) {
460 HTTPClient http;
461 String url = "http://" + ip + "/" + apiArg;
462 if (route.length() > 0) url += "?" + urlEncode(route);
463 http.begin(url);
464
465 int httpResponseCode = http.GET();
466 if(httpResponseCode < 0) Serial.println("Failed to connect to the specified IP. Is the IP correct, and is the system running?");
467
468 String respStr = extractResponse(http.getString());
469 http.end();
470
471 if (respStr == "0" || respStr == "false") return false;
472 if (respStr == "1" || respStr == "true") return true;
473
474
475 return httpResponseCode > 0; //fallback to simple http response code check
476}
